def recommend_for_pair(a_cat, a_label, a_sev, b_cat, b_label, b_sev):
    """Generate urban planning recommendations based on cross-metric analysis"""
    maxsev = max(a_sev, b_sev)
    
    # Housing/Green correlation analysis
    if a_cat == 'housing' and b_cat == 'green':
        if a_sev==2 and b_sev>=1:
            return 'High built area and low vegetation ‚Äî prioritise greening (street trees, pocket parks, urban canopy).'
    
    # Heat/Population correlation analysis  
    if a_cat in ('housing','population') and b_cat == 'heat':
        heat_sev = a_sev if a_cat == 'heat' else b_sev
        if heat_sev >= 2:
            return 'High exposure to heat in populated areas ‚Äî implement cooling measures (shading, cool roofs, tree planting).'
    
    # Air quality/Population analysis
    if a_cat in ('population','housing') and b_cat == 'air':
        air_sev = a_sev if a_cat == 'air' else b_sev
        if air_sev >= 2:
            return 'High population with poor air quality ‚Äî investigate pollution sources and protect residents.'

Purpose: Performs cross-metric correlation analysis between different NASA datasets to identify urban planning challenges. Uses semantic analysis to determine if high values are good (vegetation) or bad (pollution, heat), then provides severity-based recommendations for urban planners.

def fit_and_predict(series):
    """Generate predictions for 2026, 2028, 2030 using linear/quadratic models"""
    years = sorted(list(series.keys()))
    x = np.array(years, dtype=float)
    y = np.array([series[y] for y in years], dtype=float)
    
    # Linear regression
    coef1 = np.polyfit(x, y, 1)
    slope1, intercept1 = float(coef1[0]), float(coef1[1])
    yhat1 = slope1 * x + intercept1
    r2_1 = compute_r2(y, yhat1)
    
    # Quadratic if 3+ data points
    if len(x) >= 3:
        coef2 = np.polyfit(x, y, 2)
        a, b, c = float(coef2[0]), float(coef2[1]), float(coef2[2])
        yhat2 = a * x**2 + b * x + c
        r2_2 = compute_r2(y, yhat2)
    
    # Choose best model by R¬≤
    best_model = 'linear' if r2_1 >= r2_2 else 'quadratic'
    
    # Generate predictions for 2026, 2028, 2030
    preds = {ty: float(slope1 * ty + intercept1) for ty in [2026, 2028, 2030]}

Purpose: Generates future predictions for urban planning using statistical modeling. Processes historical NASA satellite data (2018-2024) to predict environmental trends for 2026, 2028, and 2030. Outputs predictions with confidence metrics and trend analysis for long-term urban planning decisions.
